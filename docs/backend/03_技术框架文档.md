# 诗境通 · 技术框架文档 v1.0
> React + TypeScript + Supabase + Claude API · 轻量快捷稳定技术栈

## 一、技术栈总览

### 1.1 选型原则
- **轻量**：避免过重框架，优先选择集成度高、配置少的工具
- **AI友好**：选择文档完善、AI代码生成准确率高的技术
- **稳定**：选用主流活跃社区支持的库，减少踩坑风险
- **快速**：Supabase提供开箱即用的后端，无需自建服务器

### 1.2 核心技术栈

## 四、Claude API 封装

### 4.1 基础封装（src/lib/claude.ts）

```typescript
import Anthropic from '@anthropic-ai/sdk'

const client = new Anthropic({
  apiKey: import.meta.env.VITE_ANTHROPIC_API_KEY,
  dangerouslyAllowBrowser: true, // 竞赛版前端直调，生产须走后端
})

export const MODEL = 'claude-sonnet-4-20250514'

// 普通调用（非流式）
export async function callClaude(
  systemPrompt: string,
  userMessage: string,
  maxTokens = 2000
): Promise<string> {
  const response = await client.messages.create({
    model: MODEL,
    max_tokens: maxTokens,
    system: systemPrompt,
    messages: [{ role: 'user', content: userMessage }],
  })
  return response.content[0].type === 'text' ? response.content[0].text : ''
}

// 流式调用（带回调）
export async function callClaudeStream(
  systemPrompt: string,
  messages: Array<{ role: 'user' | 'assistant'; content: string }>,
  onChunk: (text: string) => void,
  onComplete?: () => void,
  maxTokens = 3000
): Promise<void> {
  const stream = await client.messages.create({
    model: MODEL,
    max_tokens: maxTokens,
    system: systemPrompt,
    messages,
    stream: true,
  })

  for await (const event of stream) {
    if (
      event.type === 'content_block_delta' &&
      event.delta.type === 'text_delta'
    ) {
      onChunk(event.delta.text)
    }
  }

  onComplete?.()
}

// 带对话历史的流式调用（用于对话功能）
export async function chatStream(
  systemPrompt: string,
  history: Array<{ role: 'user' | 'assistant'; content: string }>,
  newMessage: string,
  onChunk: (text: string) => void,
  onComplete?: () => void
): Promise<void> {
  const messages = [
    ...history,
    { role: 'user' as const, content: newMessage },
  ]
  return callClaudeStream(systemPrompt, messages, onChunk, onComplete)
}
```

### 6.1 AI解析 Hook（src/hooks/useAnalysis.ts）

```typescript
import { useState, useCallback } from 'react'
import { callClaudeStream } from '@/lib/claude'
import { ANALYSIS_SYSTEM_PROMPT } from '@/lib/prompts'
import { supabase } from '@/lib/supabase'
import type { AnalysisResult } from '@/types/analysis'

export function useAnalysis(poemId: string) {
  const [analysis, setAnalysis] = useState<AnalysisResult | null>(null)
  const [isLoading, setIsLoading] = useState(false)
  const [streamText, setStreamText] = useState('')

  const analyze = useCallback(async (poemContent: string, depth = 'standard') => {
    setIsLoading(true)
    setStreamText('')

    // 1. 先查缓存
    const { data: cached } = await supabase
      .from('analysis_cache')
      .select('content')
      .eq('poem_id', poemId)
      .eq('depth', depth)
      .single()

    if (cached) {
      setAnalysis(cached.content as AnalysisResult)
      setIsLoading(false)
      return
    }

    // 2. 调用AI（流式）
    let fullText = ''
    await callClaudeStream(
      ANALYSIS_SYSTEM_PROMPT,
      [{ role: 'user', content: `请分析这首诗：\n${poemContent}` }],
      (chunk) => {
        fullText += chunk
        setStreamText(fullText)
      },
      async () => {
        // 3. 解析JSON并保存
        try {
          const result = JSON.parse(fullText) as AnalysisResult
          setAnalysis(result)

          // 4. 存入缓存
          await supabase.from('analysis_cache').insert({
            poem_id: poemId,
            depth,
            content: result,
          })
        } catch (e) {
          console.error('解析JSON失败', e)
        }
        setIsLoading(false)
      }
    )
  }, [poemId])

  return { analysis, isLoading, streamText, analyze }
}
```

### 6.2 对话 Hook（src/hooks/useChat.ts）

```typescript
import { useState, useCallback } from 'react'
import { chatStream } from '@/lib/claude'
import { QA_SYSTEM_PROMPT, poetRolePrompt } from '@/lib/prompts'

export type ChatMode = 'qa' | 'poet'
export type Message = { role: 'user' | 'assistant'; content: string }

export function useChat(
  poemTitle: string,
  poemContent: string,
  mode: ChatMode,
  poetKey?: string
) {
  const [messages, setMessages] = useState<Message[]>([])
  const [isStreaming, setIsStreaming] = useState(false)
  const [streamingContent, setStreamingContent] = useState('')

  const systemPrompt = mode === 'qa'
    ? QA_SYSTEM_PROMPT(poemTitle, poemContent)
    : poetRolePrompt(poetKey || 'libai', poemContent)

  const sendMessage = useCallback(async (userMessage: string) => {
    const newUserMsg: Message = { role: 'user', content: userMessage }
    const updatedMessages = [...messages, newUserMsg]
    setMessages(updatedMessages)
    setIsStreaming(true)
    setStreamingContent('')

    let aiResponse = ''
    await chatStream(
      systemPrompt,
      messages, // 历史（不含最新用户消息）
      userMessage,
      (chunk) => {
        aiResponse += chunk
        setStreamingContent(aiResponse)
      },
      () => {
        setMessages(prev => [
          ...prev,
          { role: 'assistant', content: aiResponse }
        ])
        setStreamingContent('')
        setIsStreaming(false)
      }
    )
  }, [messages, systemPrompt])

  const clearHistory = () => {
    setMessages([])
    setStreamingContent('')
  }

  return { messages, isStreaming, streamingContent, sendMessage, clearHistory }
}
```

### 6.3 弱点诊断 Hook（src/hooks/useWeakness.ts）

```typescript
import { useEffect, useState } from 'react'
import { supabase } from '@/lib/supabase'
import type { WeaknessProfile } from '@/types/weakness'

export function useWeakness(userId: string) {
  const [profile, setProfile] = useState<WeaknessProfile | null>(null)

  useEffect(() => {
    fetchProfile()
  }, [userId])

  const fetchProfile = async () => {
    const { data } = await supabase
      .from('weakness_profiles')
      .select('*')
      .eq('user_id', userId)
      .single()
    if (data) setProfile(data as WeaknessProfile)
  }

  // 每次答题后更新弱点（在答题完成时调用）
  const updateAfterAnswer = async (
    questionType: string,
    isCorrect: boolean,
    dynasty: string,
    theme: string
  ) => {
    // 获取最新profile
    const { data } = await supabase
      .from('weakness_profiles')
      .select('*')
      .eq('user_id', userId)
      .single()

    const current = data || {
      user_id: userId,
      by_question_type: {},
      by_dynasty: {},
      by_theme: {},
    }

    // 更新题型维度
    const qt = current.by_question_type[questionType] || { attempts: 0, correct: 0 }
    qt.attempts += 1
    if (isCorrect) qt.correct += 1
    qt.rate = qt.correct / qt.attempts
    current.by_question_type[questionType] = qt

    // 更新朝代维度
    const dy = current.by_dynasty[dynasty] || { attempts: 0, correct: 0 }
    dy.attempts += 1
    if (isCorrect) dy.correct += 1
    dy.rate = dy.correct / dy.attempts
    current.by_dynasty[dynasty] = dy

    // 更新主题维度
    const th = current.by_theme[theme] || { attempts: 0, correct: 0 }
    th.attempts += 1
    if (isCorrect) th.correct += 1
    th.rate = th.correct / th.attempts
    current.by_theme[theme] = th

    // 计算弱点维度（rate < 0.7的视为弱点）
    const weakDimensions = Object.entries(current.by_question_type)
      .filter(([_, v]: any) => v.rate < 0.7 && v.attempts >= 3)
      .sort(([_, a]: any, [__, b]: any) => a.rate - b.rate)
      .map(([k]) => k)

    current.weak_dimensions = weakDimensions

    // Upsert
    await supabase.from('weakness_profiles').upsert({
      ...current,
      last_updated: new Date().toISOString(),
    })

    setProfile(current as WeaknessProfile)
  }

  // 雷达图数据格式化
  const radarData = profile ? [
    { subject: '默写', score: (profile.by_question_type['dictation']?.rate || 0) * 100 },
    { subject: '词义', score: (profile.by_question_type['meaning']?.rate || 0) * 100 },
    { subject: '手法', score: (profile.by_question_type['technique']?.rate || 0) * 100 },
    { subject: '情感', score: (profile.by_question_type['emotion']?.rate || 0) * 100 },
    { subject: '赏析', score: (profile.by_question_type['analysis']?.rate || 0) * 100 },
    { subject: '活用', score: (profile.by_question_type['context']?.rate || 0) * 100 },
  ] : []

  return { profile, radarData, updateAfterAnswer, fetchProfile }
}
```

```typescript
    }

    }
```

## 九、SM-2 间隔重复算法（src/lib/sm2.ts）

```typescript
// 基于 SM-2 算法计算下次复习时间
// quality: 0=完全忘记, 1=严重错误, 2=困难回忆, 3=一定困难, 4=完美, 5=轻松完美

export interface SM2State {
  interval: number    // 距今天数
  repetitions: number // 重复次数
  easeFactor: number  // 难易系数 (默认2.5)
}

export function calculateNextReview(
  quality: number, // 0-5
  state: SM2State
): SM2State & { nextDate: Date } {
  let { interval, repetitions, easeFactor } = state

  if (quality < 3) {
    // 回答不好，重新开始
    interval = 1
    repetitions = 0
  } else {
    if (repetitions === 0) {
      interval = 1
    } else if (repetitions === 1) {
      interval = 6
    } else {
      interval = Math.round(interval * easeFactor)
    }
    repetitions += 1
  }

  // 更新难易系数
  easeFactor = Math.max(1.3,
    easeFactor + 0.1 - (5 - quality) * (0.08 + (5 - quality) * 0.02)
  )

  const nextDate = new Date()
  nextDate.setDate(nextDate.getDate() + interval)

  return { interval, repetitions, easeFactor, nextDate }
}

// 将记忆质量映射到0-5
export function isCorrectToQuality(isCorrect: boolean, timeSpent: number): number {
  if (!isCorrect) return 1
  if (timeSpent < 5) return 5   // 5秒内答对
  if (timeSpent < 15) return 4  // 15秒内答对
  return 3                       // 超过15秒答对
}
```

### 11.3 常见踩坑预防

| 问题 | 预防方案 |
|-----|---------|
| Claude API 流式输出在React中状态更新混乱 | 使用 useRef 存储 aiResponse，onComplete 时统一 setState |
| Supabase RLS 权限报错 | 确保每张表都有对应 Policy，用 supabase.auth.getUser() 验证 |
| 流式文字出现乱码 | 确保所有字符串拼接在 onChunk 回调中，不做异步处理 |
| TypeScript 类型与 Supabase 响应不匹配 | 使用 as 断言，或生成 Supabase TypeScript 类型 |
| 诗词中文搜索不准确 | 使用 Supabase 的 ilike 操作符进行模糊搜索 |
| Recharts 在 SSR 报错 | 确保图表组件用 dynamic import 或在 useEffect 中渲染 |
